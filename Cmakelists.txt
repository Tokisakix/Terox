cmake_minimum_required(VERSION 3.14)
project(_C)

add_subdirectory(
    ${CMAKE_SOURCE_DIR}/extern/pybind11
)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include_directories(
    ${CMAKE_SOURCE_DIR}/aten/include
)

file(GLOB SOURCES
    "aten/src/*.cpp"
)

pybind11_add_module(_C ${SOURCES})

install(TARGETS _C DESTINATION ${CMAKE_SOURCE_DIR}/terox/)
